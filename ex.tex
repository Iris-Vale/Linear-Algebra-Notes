\section{矩阵的相抵与相似}

\begin{example}
    ~

    (1)~~求证：若$n$阶方阵$A$是幂等矩阵（$A^2 = A$），则$A$必可对角化.

    (2)~~求证：若$n$阶方阵$A$是对合矩阵（$A^2 = I_n$），则$A$必可对角化.
\end{example}

~\par

\begin{example}
    设$n$阶矩阵：
    \begin{eq1}
        A = 
        \begin{pmatrix}
            I_r & B \\[2pt]
            O & -I_{n-r}
        \end{pmatrix}\end{eq1}

    求证：$A$可对角化.
\end{example}

~\par

\begin{example}
    设$A,B$为$n$阶方阵. 
    求证：$AB-BA$必定不相似于$kI_n~(k \ne 0)$.
\end{example}

~\par

\begin{example}
    设$\mathbb{C}$上的$n$阶矩阵$A,B$有相同的特征值，且这$n$个特征值互不相等.
    求证：存在$\mathbb{C}$上的$n$阶矩阵$P,Q$，使得$A=PQ$，$B=QP$.
\end{example}
